* Theme Helpers :noexport:
#+NAME: get-val
#+begin_src emacs-lisp :var key="font-family" :exports none :tangle no
  (with-current-buffer (find-file-noselect "../theme.org")
    (save-excursion
      (goto-char (point-min))
      (re-search-forward "^#\\+NAME: theme-table")
      (forward-line)
      (let* ((full-table (org-babel-read-table))
             (headers (car full-table))
             (table (cl-remove-if (lambda (x) (or (symbolp x) (not (listp x)))) (cdr full-table)))
             (host (system-name))
             (col-idx (or (cl-position host headers :test 'string-equal) 1))
             (row (assoc key table))
             (val (nth col-idx row))
             (val-str (format "%s" val)))
        (if (or (null val) (string-equal val-str "") (string-equal val-str "-") (string-equal val-str "nil"))
            (nth 1 row)
          val))))
#+end_src

#+NAME: get-color
#+begin_src emacs-lisp :var name="bg" nohash="nil" :exports none :tangle no
  (let ((color (with-current-buffer (find-file-noselect "../theme.org")
                 (save-excursion
                   (goto-char (point-min))
                   (re-search-forward "^#\\+NAME: palette")
                   (forward-line)
                   (let* ((full-table (org-babel-read-table))
                          (headers (car full-table))
                          (table (cl-remove-if (lambda (x) (or (symbolp x) (not (listp x)))) (cdr full-table)))
                          (host (system-name))
                          (col-idx (or (cl-position host headers :test 'string-equal) 1))
                          (row (assoc name table))
                          (val (nth col-idx row))
                          (val-str (format "%s" val)))
                     (if (or (null val) (string-equal val-str "") (string-equal val-str "-") (string-equal val-str "nil"))
                         (nth 1 row)
                       val))))))
    (if (and color (string-equal nohash "t"))
        (replace-regexp-in-string "^#" "" color)
      color))
#+end_src

#+begin_src bash :tangle "~/.local/bin/osd.sh" :tangle-mode (identity #o777)
#!/bin/bash

MESSAGE=$1
ICON_NAME=$2
ICON_PATH="/usr/share/icons/Adwaita/symbolic/status/${ICON_NAME}.svg"

# Fallback to Legacy PNG
if [ ! -f "$ICON_PATH" ]; then
    ICON_PATH="/usr/share/icons/AdwaitaLegacy/32x32/legacy/${ICON_NAME}.png"
fi

makoctl dismiss -a "osd"

# THE GDBUS CALL
# Arguments: (app_name, replace_id, app_icon, summary, body, actions, hints, expire_timeout)
gdbus call --session \
    --dest org.freedesktop.Notifications \
    --object-path /org/freedesktop/Notifications \
    --method org.freedesktop.Notifications.Notify \
    "osd" 999 "$ICON_PATH" "$MESSAGE" "" "[]" \
    "{\"x-mako-stack-tag\": <'osd-notification'>, \"transient\": <true>}" 500 > /dev/null

#+end_src
