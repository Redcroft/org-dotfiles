  local function get_hostname()
     local handle = io.popen("hostname")

     if not handle then
        return "unknown"
     end

     local result = handle:read("*l")
     handle:close()

     return result or "unknown"
  end

  function show_osd(message, icon_name)
     local icon_path = "/usr/share/icons/Adwaita/symbolic/status/" .. icon_name .. ".svg"
     
     local f = io.open(icon_path, "r")
     if f then 
        f:close() 
     else
        icon_path = "/usr/share/icons/AdwaitaLegacy/32x32/legacy/" .. icon_name .. ".png"
     end

     hl.exec_cmd("makoctl dismiss -a 'osd'")
     
     local gdbus_cmd = string.format(
        [[gdbus call --session --dest org.freedesktop.Notifications ]] ..
        [[--object-path /org/freedesktop/Notifications ]] ..
        [[--method org.freedesktop.Notifications.Notify ]] ..
        [["osd" 999 "%s" "%s" "" "[]" ]] ..
        [[ '{"x-mako-stack-tag": <"osd-notification">, "transient": <true>}' 500]],
        icon_path, message
     )

     hl.exec_cmd(gdbus_cmd)
  end

  local function change_brightness(cmd_type, action, value)
     -- 1. Execute the monitor command
     local monitor_call = string.format("monitorctl %s %s %s", cmd_type, action, value or "")
     os.execute(monitor_call)

     -- 2. Fetch the new value
     local handle = io.popen("monitorctl b -g")
     local current_val = "0"
     if handle then
        current_val = handle:read("*a"):gsub("%s+", "")
        handle:close()
     end

     -- 3. Show OSD
     show_osd("Brightness " .. current_val .. "%", "display-brightness-symbolic")
  end

#+begin_src conf :tangle "~/.config/hypr/hypridle.conf"
  general {
      lock_cmd = pidof hyprlock || hyprlock
      before_sleep_cmd = loginctl lock-session
      after_sleep_cmd = hyprctl dispatch dpms on
  }

  # Dim screen after 1 min
  listener {
      timeout = 60
      on-timeout = brightnessctl -s set 10
      on-resume = brightnessctl -r
  }

  # Monitor DPMS off after 3 mins
  listener {
      timeout = 180
      on-timeout = hyprctl dispatch dpms off
      on-resume = hyprctl dispatch dpms on
  }

  # This is the "Safety Net" for manual locks
  # It ensures that if the session is locked, we listen for resume events
  listener {
      timeout = 300
      on-timeout = loginctl lock-session
  }
