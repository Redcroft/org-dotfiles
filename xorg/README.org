*** Mouse
#+begin_src conf :tangle "/sudo::/etc/X11/xorg.conf.d/50-libinput.conf"
  Section "InputClass"
      Identifier "Mouse"
      Driver "libinput"
      MatchIsPointer "on"
      Option "AccelProfile" "flat"
  EndSection
#+end_src

*** Touchpad
#+begin_src conf :tangle "/sudo::/etc/X11/xorg.conf.d/50-libinput.conf"
  Section "InputClass"
      Identifier "Touchpad"
      Driver "libinput"
      MatchIsTouchpad "on"
      Option "AccelProfile" "flat"
      Option "ClickMethod" "clickfinger"
      Option "NaturalScrolling" "true"
      Option "DisableWhileTyping" "false"
      Option "Tapping" "on"
  EndSection
#+end_src
