* Resampler & Clock
Default graph rate is 48 kHz (matches the HeSuVi HRIR WAV's native rate,
removing internal resampling in the convolver), with the full set of UAC 2.0
rates allowed so the graph can switch to native rates when bypassing the WAV
filter for high-resolution sources. Resampler quality is pinned to 14 (max);
CPU cost is minor at music buffer sizes, benefit is measurable on transients.

#+begin_src conf :tangle "~/.config/pipewire/pipewire.conf.d/10-quality.conf"
  context.properties = {
      resample.quality            = 14
      link.max-buffers            = 16
      default.clock.rate          = 48000
      default.clock.allowed-rates = [ 44100 48000 88200 96000 176400 192000 ]
  }
#+end_src

* DAC Dither (heimdall)
The Fosi ZH3 runs at 24-bit over USB. Pin the alsa node to S24 so the hardware
path uses 24-bit samples, and enable 2 bits of dither noise to truncate the
float32 mix without quantization distortion. The node's audio.rate is left
unset so the DAC follows the graph rate (48k default, switching to higher
native rates for hi-res sources when bypassing the WAV filter). Disabled on
non-heimdall hosts via the host guard.

#+begin_src conf :tangle (if (string= system-name "heimdall") "~/.config/pipewire/pipewire.conf.d/51-fosi-dither.conf" "")
  monitor.alsa.rules = [
    { matches = [ { node.name = "alsa_output.usb-Fosi_Fosi_Audio_ZH3-00.analog-stereo" } ]
      actions = {
        update-props = {
          dither.noise                   = 2
          session.suspend-timeout-seconds = 0
          audio.format                   = "S24"
        }
      }
    }
  ]
#+end_src

* Headphone EQ
Parametric EQ for the Hifiman HE400se, derived from AutoEq using oratory1990's
measurement against the Harman over-ear 2018 target. Sits before the spatial
sink in the signal path (EQ playback targets the spatial sink's capture node).

The AutoEq preset calls for -6.0 dB of preamp to avoid clipping when the low
shelf at 105 Hz adds +6.7 dB. This is baked into the final mixers as a linear
gain of 0.5012 (= 10^(-6/20)), applied pre-DAC so the S24 output stage never
clips on the EQ boost. Software node volume stays at 1.0 (0 dB) — use the Fosi
ZH3's analog knob for listening level.

#+begin_src conf :tangle "~/.config/pipewire/pipewire.conf.d/98-eq-he400se.conf"
  context.modules = [
      { name = libpipewire-module-filter-chain
          args = {
              node.description = "HE400se Parametric EQ"
              media.name       = "HE400se Parametric EQ"
              filter.graph = {
                  nodes = [
                      { type = builtin label = copy   name = copyL }
                      { type = builtin label = copy   name = copyR }

                      # Band 1 - low shelf 105 Hz +6.7 dB
                      { type = builtin label = bq_lowshelf  name = eqL1  control = { "Freq" = 105   "Q" = 0.70 "Gain" =  6.7 } }
                      { type = builtin label = bq_lowshelf  name = eqR1  control = { "Freq" = 105   "Q" = 0.70 "Gain" =  6.7 } }
                      # Band 2 - peak 1968 Hz +5.1 dB
                      { type = builtin label = bq_peaking   name = eqL2  control = { "Freq" = 1968  "Q" = 2.37 "Gain" =  5.1 } }
                      { type = builtin label = bq_peaking   name = eqR2  control = { "Freq" = 1968  "Q" = 2.37 "Gain" =  5.1 } }
                      # Band 3 - peak 861 Hz -2.3 dB
                      { type = builtin label = bq_peaking   name = eqL3  control = { "Freq" = 861   "Q" = 1.86 "Gain" = -2.3 } }
                      { type = builtin label = bq_peaking   name = eqR3  control = { "Freq" = 861   "Q" = 1.86 "Gain" = -2.3 } }
                      # Band 4 - peak 101 Hz -2.4 dB
                      { type = builtin label = bq_peaking   name = eqL4  control = { "Freq" = 101   "Q" = 0.32 "Gain" = -2.4 } }
                      { type = builtin label = bq_peaking   name = eqR4  control = { "Freq" = 101   "Q" = 0.32 "Gain" = -2.4 } }
                      # Band 5 - peak 1619 Hz +1.1 dB
                      { type = builtin label = bq_peaking   name = eqL5  control = { "Freq" = 1619  "Q" = 3.53 "Gain" =  1.1 } }
                      { type = builtin label = bq_peaking   name = eqR5  control = { "Freq" = 1619  "Q" = 3.53 "Gain" =  1.1 } }

                      { type = builtin label = bq_highshelf name = eqR6  control = { "Freq" = 10000 "Q" = 0.70 "Gain" = -2.6 } }

                      { type = builtin label = bq_peaking   name = eqR7  control = { "Freq" = 138   "Q" = 1.98 "Gain" =  0.7 } }
                      # Band 8 - peak 64 Hz -0.7 dB
                      { type = builtin label = bq_peaking   name = eqL8  control = { "Freq" = 64    "Q" = 1.95 "Gain" = -0.7 } }

                      # Band 9 - peak 251 Hz -0.4 dB
                      { type = builtin label = bq_peaking   name = eqL9  control = { "Freq" = 251   "Q" = 2.33 "Gain" = -0.4 } }
                      { type = builtin label = bq_peaking   name = eqR9  control = { "Freq" = 251   "Q" = 2.33 "Gain" = -0.4 } }
                      # Band 10 - peak 29 Hz +0.5 dB
                      { type = builtin label = bq_peaking   name = eqL10 control = { "Freq" = 29    "Q" = 2.63 "Gain" =  0.5 } }
                      { type = builtin label = bq_peaking   name = eqR10 control = { "Freq" = 29    "Q" = 2.63 "Gain" =  0.5 } }

                      { type = builtin label = mixer name = mixL control = { "Gain" = 0.5012 } }
                      { type = builtin label = mixer name = mixR control = { "Gain" = 0.5012 } }
                  ]
                  links = [
                      { output = "copyL:Out"  input = "eqL1:In"  }
                      { output = "copyR:Out"  input = "eqR1:In"  }
                      { output = "eqL1:Out"   input = "eqL2:In"  }
                      { output = "eqR1:Out"   input = "eqR2:In"  }
                      { output = "eqL2:Out"   input = "eqL3:In"  }
                      { output = "eqR2:Out"   input = "eqR3:In"  }
                      { output = "eqL3:Out"   input = "eqL4:In"  }
                      { output = "eqR3:Out"   input = "eqR4:In"  }
                      { output = "eqL4:Out"   input = "eqL5:In"  }

                      { output = "eqR8:Out"   input = "eqR9:In"  }
                      { output = "eqL9:Out"   input = "eqL10:In" }
                      { output = "eqR9:Out"   input = "eqR10:In" }
                      { output = "eqL10:Out"  input = "mixL:In 1" }
                      { output = "eqR10:Out"  input = "mixR:In 1" }
                  ]
                  inputs  = [ "copyL:In" "copyR:In" ]
                  outputs = [ "mixL:Out" "mixR:Out" ]
              }
              capture.props = {
                  node.name      = "effect_input.eq-he400se"
                  media.class    = Audio/Sink
