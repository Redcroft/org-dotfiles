#+name: enc-type
#+begin_src emacs-lisp :results value :cache yes
  (defun parse-version-number (version-string)
  "Parse a version string and return the major.minor version."
  (when (string-match "Version \\([0-9]+\\.[0-9]+\\)" version-string)
    (match-string 1 version-string)))

  (setq version (string-to-number (parse-version-number (shell-command-to-string "mu --version"))))

  (if (> version 1.12)
      "TLSType"
    "SSLType")
#+end_src

** MBSync
*** Redcroft
#+begin_src conf :tangle "~/.mbsyncrc" :noweb yes
  IMAPAccount redcroft
  Host imap.hostinger.com
  Port 993
  User richard@redcroft.tech
  PassCmd "pass hostinger.com/richard@redcroft.tech"
  <<enc-type()>> IMAPS

  IMAPStore redcroft-remote
  Account redcroft

  MaildirStore redcroft-local
  SubFolders Verbatim
  Path ~/Mail/Redcroft/
  Inbox ~/Mail/Redcroft/Inbox
  Trash ~/Mail/Redcroft/Trash

  Channel redcroft
  Far :redcroft-remote:
  Near :redcroft-local:
  Patterns *
  SyncState *
  Create Both
  Expunge Both
  CopyArrivalDate yes
  Sync All
#+end_src
